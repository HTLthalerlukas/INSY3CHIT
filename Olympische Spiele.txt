create table helfer
(
    Personalnummer int         not null
        primary key,
    Name           varchar(45) not null
);

create table nationen
(
    Nation            varchar(45) not null
        primary key,
    Mannschaftsleiter varchar(45) not null,
    Telefonnummer     varchar(45) not null
);

create table schiedsrichter
(
    Personalnummer  int         not null
        primary key,
    Name            varchar(45) not null,
    nationen_Nation varchar(45) not null,
    constraint fk_schiedsrichter_nationen1
        foreign key (nationen_Nation) references nationen (Nation)
);

create index fk_schiedsrichter_nationen1_idx
    on schiedsrichter (nationen_Nation);

create table sportler
(
    Startnummer     int         not null
        primary key,
    Name            varchar(45) not null,
    nationen_Nation varchar(45) not null,
    constraint fk_sportler_nationen1
        foreign key (nationen_Nation) references nationen (Nation)
);

create index fk_sportler_nationen1_idx
    on sportler (nationen_Nation);

create table wettkämpfe
(
    Wettkampfid                   int         not null
        primary key,
    Sportart                      varchar(45) not null,
    Termin                        varchar(45) not null,
    Sportstätte                   varchar(45) not null,
    schiedsrichter_Personalnummer int         not null,
    constraint fk_wettkämpfe_schiedsrichter1
        foreign key (schiedsrichter_Personalnummer) references schiedsrichter (Personalnummer)
);

create index fk_wettkämpfe_schiedsrichter1_idx
    on wettkämpfe (schiedsrichter_Personalnummer);

create table wettkämpfe_has_helfer
(
    wettkämpfe_Wettkampfid int not null,
    helfer_Personalnummer  int not null,
    primary key (wettkämpfe_Wettkampfid, helfer_Personalnummer),
    constraint fk_wettkämpfe_has_helfer_helfer1
        foreign key (helfer_Personalnummer) references helfer (Personalnummer),
    constraint fk_wettkämpfe_has_helfer_wettkämpfe1
        foreign key (wettkämpfe_Wettkampfid) references wettkämpfe (Wettkampfid)
);

create index fk_wettkämpfe_has_helfer_helfer1_idx
    on wettkämpfe_has_helfer (helfer_Personalnummer);

create index fk_wettkämpfe_has_helfer_wettkämpfe1_idx
    on wettkämpfe_has_helfer (wettkämpfe_Wettkampfid);

create table wettkämpfe_has_sportler
(
    wettkämpfe_Wettkampfid int not null,
    sportler_Startnummer   int not null,
    primary key (wettkämpfe_Wettkampfid, sportler_Startnummer),
    constraint fk_wettkämpfe_has_sportler_sportler1
        foreign key (sportler_Startnummer) references sportler (Startnummer),
    constraint fk_wettkämpfe_has_sportler_wettkämpfe
        foreign key (wettkämpfe_Wettkampfid) references wettkämpfe (Wettkampfid)
);

create index fk_wettkämpfe_has_sportler_sportler1_idx
    on wettkämpfe_has_sportler (sportler_Startnummer);

create index fk_wettkämpfe_has_sportler_wettkämpfe_idx
    on wettkämpfe_has_sportler (wettkämpfe_Wettkampfid);
